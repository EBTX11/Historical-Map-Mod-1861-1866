                        holder = c:FRA
                        type = revoke_claim
                        target_state = s:STATE_ALSACE_LORRAINE.region_state:FRA  
                        target_country = c:GER
                    }  

                    add_war_goal = {
                        holder = c:GER 
                        type = humiliation
                        target_country = c:FRA
                        #target_state = s:STATE_WESTERN_SERBIA.region_state:TUR
